13  /**
14   * A class used by the GUI to update the database.
15   */
16  public abstract class DBMethods {
17  
18    /**
19     * Method to delete 'toRemove' from 'item'.
20     * @param wrType the type of toRemove, and thus the reverse of item
21     * @param item the movie/actor we're removing from
22     * @param toRemove the actor/movie we're removing
23     * @param db database
24     */
25    public static void delete(String wrType, String item,
26                              String toRemove, Db db) {
27      Entity ent;
28      try {
29        Connection conn = db.getConnection();
30        PreparedStatement prep = conn.prepareStatement(
31                "DELETE FROM actor_film WHERE actor = ? AND film = ?;");
32        if (wrType.equals("actor")) {
33          ent = MovieProxy.ofName(item, db);
34          prep.setString(1, ActorProxy.ofName(toRemove, db).getId());
35          prep.setString(2, ent.getId());
36        } else {
37          ent = ActorProxy.ofName(item, db);
38          prep.setString(1, ent.getId());
39          prep.setString(2, MovieProxy.ofName(toRemove, db).getId());
40        }
41  
42        try {
43          prep.executeUpdate();
44        } catch (SQLException se) {
45          throw new RuntimeException(se);
46        }
47      } catch (SQLException se) {
48        throw new RuntimeException(se);
49      }
50  
51      EntityProxy.deleteFrom(ent.getId());
52    }
53  
54    /**
55     *  Method to change the name of 'item'.
56     * @param wrType the opposite type of item (actor goes to film)
57     * @param item the entitiy we're changing
58     * @param name the name we're changing it to
59     * @param db db connection
60     */
61    public static void change(String wrType, String item, String name, Db db) {
62      Entity ent;
63      String database;
64      if (wrType.equals("actor")) {
65        ent = MovieProxy.ofName(item, db);
66        database = "film";
67      } else {
68        ent = ActorProxy.ofName(item, db);
69        database = "actor";
70      }
71      try {
72        Connection conn = db.getConnection();
73        PreparedStatement prep = conn.prepareStatement(
74                "UPDATE " + database + " SET name = ? WHERE id = ?;");
75        prep.setString(1, name);
76        prep.setString(2, ent.getId());
77  
78        try {
79          prep.executeUpdate();
80          prep.close();
81        } catch (SQLException se) {
82          throw new RuntimeException(se);
83        }
84      } catch (SQLException se) {
85        throw new RuntimeException(se);
86      }
87        /*Ideally we'd delete only the entities affected from the cache, but I
88        believes that it's not trivial to determine.
89         */
90      EntityProxy.deleteAll();
91    }
92  
93    /**
94     * Method to add a new entity 'name' to item. If it exists, 'name' is added.
95     * Otherwise we construct add and add it.
96     * @param wrType the type name, and opposite type of item
97     * @param item the item we're adding to
98     * @param name what we're item.
99     * @param db the database we're using.
100    */
101   public static void add(String wrType, String item, String name, Db db) {
102     Entity a;
103     Entity b;
104     String actor;
105     String film;
106     if (wrType.equals("actor")) {
107       a = MovieProxy.ofName(item, db);
108       b = ActorProxy.ofName(name, db);
109       if (b == null) {
110         createNewEntry("actor", name, db);
111         b = ActorProxy.ofName(name, db);
112       }
113       actor = b.getId();
114       film = a.getId();
115     } else {
116       a = ActorProxy.ofName(item, db);
117       b = MovieProxy.ofName(name, db);
118       if (b == null) {
119         createNewEntry("film", name, db);
120         b = MovieProxy.ofName(name, db);
121       }
122       actor = a.getId();
123       film = b.getId();
124     }
125     try {
126       Connection conn = db.getConnection();
127       PreparedStatement prep = conn.prepareStatement(
128               "INSERT INTO actor_film VALUES(?, ?);");
129       prep.setString(1, actor);
130       prep.setString(2, film);
131 
132 
133       try {
134         prep.executeUpdate();
135         prep.close();
136       } catch (SQLException se) {
137         throw new RuntimeException(se);
138       }
139     } catch (SQLException se) {
140       throw new RuntimeException(se);
141     }
142 
143     EntityProxy.deleteAll();
144   }
145   /**
146    * Method to create a new entity 'name' of type 'type'.
147    *
148    * @param type the type, either actor or film
149    * @param name the name of the new entity
150    * @param db the database connection
151    */
152   public static void createNewEntry(String type, String name, Db db) {
153     try {
154       Connection conn = db.getConnection();
155       PreparedStatement prep = conn.prepareStatement(
156               "INSERT INTO " + type + " VALUES(?, ?);");
157       UUID uid = UUID.randomUUID();
158       prep.setString(1, uid.toString());
159       prep.setString(2, name);
160 
161       try {
162         prep.executeUpdate();
163         prep.close();
164       } catch (SQLException se) {
165         throw new RuntimeException(se);
166       }
167     } catch (SQLException se) {
168       throw new RuntimeException(se);
169     }
170 
171     EntityProxy.deleteAll();
172   }
