10  public abstract class Autocorrect {
11    /**
12     * Takes in a line of text and autocorrects the last word.
13     *
14     * @param line the line we are looking at for corrections
15     * @param wd the worddata that stores information
16     * @return List of up to five corrections to the final word
17     */
18    public static List<String> autocorrect(String line, WordData wd) {
19      List<String> answers = new ArrayList<String>();
20      if (!WordData.getCapsAndLetters().contains(
21              line.substring(line.length() - 1, line.length()))) {
22        return answers;
23      }
24      List<String> parts = new ArrayList<>();
25      WordData.parseLine(parts, line);
26      if (parts.size() < 1) {
27        System.out.println("ERROR: no arguements given");
28        return answers;
29      }
30      String word = parts.get(parts.size() - 1);
31      String preword = "";
32      if (parts.size() > 1) {
33        preword = parts.get(parts.size() - 2);
34      }
35      Command command = new Command(wd, answers,
36              word, preword, false);
37      command.exact();
38      command.levenshtein(1);
39      command.prefix();
40      command.whitespace();
41      return answers;
42    }
43  }
