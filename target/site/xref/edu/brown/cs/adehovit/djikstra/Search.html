6   import java.util.PriorityQueue;

11  /** Djikstra search implementation.
12   *
13   * @author Adam DeHovitz
14   */
15  public abstract class Search {
16  
17    /**
18     * Given a start and end goal, finds the path (if it exists) between them.
19     * Implements Djikstra's algorithm
20     *
21     * @param p      Name of the starting actor
22     * @param goalID id of the ending actor
23     * @param <T> a generic that implements Node
24     * @return A list that describes each part of the path to the end
25     */
26    public static <T extends Node> List<List<String>> dijkstra(T p, String goalID)
27    {
28      Set<String> visited = new HashSet<>();
29      PriorityQueue<Node> queue = new PriorityQueue<>();
30      queue.add(p);
31      while (queue.size() > 0) {
32        //Getting our current node
33        p = (T) queue.poll();
34        assertTrue(p != null);
35        if (p.getId().equals(goalID)) {
36          //we found our target
37          return p.getPath();
38        }
39        if (!visited.contains(p.getId())) {
40          //let's add new nodes
41          visited.add(p.getId());
42          p.addNeighbors(queue);
43        }
44      }
45      return new ArrayList<>();
46    }
47  }
